from flask import Flask, request, jsonify
from flask_cors import CORS
import json

app = Flask(__name__)
CORS(app)  # Permite conexiones seguras

# Base de datos ligera en formato JSON nativo
VACANTES_DATABASE = [
    {
        "id": 1,
        "title": "Senior IT Project Manager (Remoto)",
        "company": "Tech Solutions Global",
        "lang": "es",
        "matchScore": 94,
        "desc": "Liderar implementaciones tecnológicas, Scrum/Kanban."
    },
    {
        "id": 2,
        "title": "Cloud Infrastructure Specialist",
        "company": "DataCorp International",
        "lang": "en",
        "matchScore": 78,
        "desc": "Design and implement AWS cloud solutions."
    }
]

@app.route('/api/buscar', methods=['GET'])
def buscar_vacantes():
    query = request.args.get('q', '').lower()
    idioma = request.args.get('lang', 'all')
    
    resultados = VACANTES_DATABASE
    
    # Filtrado ultra rápido en el servidor con Python
    if idioma != 'all':
        resultados = [v for v in resultados if v['lang'] == idioma]
        
    if query:
        resultados = [v for v in resultados if query in v['title'].lower() or query in v['desc'].lower()]
        
    return jsonify(resultados)

if __name__ == '__main__':
    app.run(port=5000, debug=True)
